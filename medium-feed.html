<script>

(async () => {

  const HANDLE = "brianiselin67";   // your Medium handle

  const MAX = 12;                    // how many posts to show

  const url = "https://r.jina.ai/http://medium.com/feed/@" + HANDLE; // simple CORS-safe proxy

  try {

    const xml = await (await fetch(url, {cache:"no-store"})).text();

    const doc = new DOMParser().parseFromString(xml, "text/xml");

    const items = [...doc.querySelectorAll("item")].slice(0, MAX).map(it => {

      const pick = t => it.querySelector(t)?.textContent || "";

      const enc = it.getElementsByTagNameNS("http://purl.org/rss/1.0/modules/content/","encoded")[0]?.textContent || "";

      const img = (enc.match(/<img[^>]+src="([^"]+)"/i) || [])[1] || null;

      const words = enc.replace(/<[^>]+>/g," ").trim().split(/\s+/).filter(Boolean).length;

      return { title: pick("title"), link: pick("link"), pubDate: pick("pubDate"),

        author: it.getElementsByTagNameNS("http://purl.org/dc/elements/1.1/","creator")[0]?.textContent || "",

        image: img, readMin: Math.max(2, Math.round(words/200)) };

    });

    const fmt = d => new Date(d).toLocaleDateString(undefined,{year:'numeric',month:'short',day:'numeric'});

    const card = p => `

      <article>

        ${p.image?`<a class="img" href="${p.link}" target="_blank" rel="noopener"><img src="${p.image}" alt=""></a>`:""}

        <div class="card">

          <a href="${p.link}" target="_blank" rel="noopener"><h3>${p.title}</h3></a>

          <div class="meta">${[p.pubDate?fmt(p.pubDate):"", p.readMin?`${p.readMin} min read`:"", p.author].filter(Boolean).join(" • ")}</div>

          <a class="link" href="${p.link}" target="_blank" rel="noopener">Read on Medium →</a>

        </div>

      </article>`;

    grid.innerHTML = items.map(card).join("");

  } catch(e){ console.error(e); err.style.display = "block"; }

})();

</script>
